def prefix(s):
    p = [0]*len(s)
    for i in range(len(s)-1):
        j = p[i]
        while j>0 and s[j]!=s[i+1]:
            j=p[j-1]
        if s[j]==s[i+1]:
            p[i+1]= j+1
        else:
            p[i+1]=0
    return p

s = input()
s_t = s+"$"+s 
p = prefix(s_t)
p0=p[-1]-p[len(s)-1]
a = len(s)//p0
if s[:p0]*a == s:
    print(a)
else:
    print(1)
